// Initialize jsPDF const { jsPDF } = window.jspdf; // DOM Elements const regionSelect = document.getElementById('region-select'); const auditMode = document.getElementById('audit-mode'); // Populate country dropdown function loadCountries() { const countries = AioxyGHGData.getCountries(); countries.forEach(country => { const option = document.createElement('option'); option.value = country; option.textContent = country; regionSelect.appendChild(option); }); } // Update factors when region changes regionSelect.addEventListener('change', function() { const region = this.value; if (!region) return; // Scope 1 updateFactorDisplay('petrol', region, 'scope1', 'fuel'); updateFactorDisplay('diesel', region, 'scope1', 'fuel'); // Scope 2 updateFactorDisplay('electricity', region, 'scope2', 'electricity'); // Scope 3 updateFactorDisplay('petrol-wtt', region, 'scope3', 'fuel_wtt'); updateFactorDisplay('diesel-wtt', region, 'scope3', 'fuel_wtt'); calculateAll(); }); // Toggle audit sources auditMode.addEventListener('change', function() { calculateAll(); }); // Update factor displays function updateFactorDisplay(elementPrefix, region, scope, category) { const type = elementPrefix.includes('wtt') ? elementPrefix.split('-')[0] : elementPrefix; const data = AioxyGHGData.getFactor(region, scope, category)[type]; document.getElementById(`${elementPrefix}-factor`).textContent = data.factor; if (auditMode.checked) { document.getElementById(`${elementPrefix}-source`).innerHTML = `| Source: ${data.source}`; } else { document.getElementById(`${elementPrefix}-source`).textContent = ''; } } // Main calculation function function calculateAll() { const region = regionSelect.value; if (!region) return; // Scope 1 calculateEmission('petrol', region, 'scope1', 'fuel'); calculateEmission('diesel', region, 'scope1', 'fuel'); const scope1Total = parseFloat(document.getElementById('petrol-result').textContent) + parseFloat(document.getElementById('diesel-result').textContent); document.getElementById('scope1-total').textContent = scope1Total.toFixed(2); // Scope 2 calculateEmission('electricity', region, 'scope2', 'electricity'); const scope2Total = parseFloat(document.getElementById('electricity-result').textContent); document.getElementById('scope2-total').textContent = scope2Total.toFixed(2); // Scope 3 calculateEmission('petrol-wtt', region, 'scope3', 'fuel_wtt'); calculateEmission('diesel-wtt', region, 'scope3', 'fuel_wtt'); const scope3Total = parseFloat(document.getElementById('petrol-wtt-result').textContent) + parseFloat(document.getElementById('diesel-wtt-result').textContent); document.getElementById('scope3-total').textContent = scope3Total.toFixed(2); // Grand Total const total = scope1Total + scope2Total + scope3Total; document.getElementById('total-emissions').textContent = total.toFixed(2); } function calculateEmission(inputId, region, scope, category) { const type = inputId.includes('wtt') ? inputId.split('-')[0] : inputId; const inputElement = document.getElementById(inputId); const resultElement = document.getElementById(`${inputId}-result`); const quantity = parseFloat(inputElement.value) || 0; const factorData = AioxyGHGData.getFactor(region, scope, category)[type]; resultElement.textContent = (quantity * factorData.factor).toFixed(2); } // PDF Export document.getElementById('export-pdf').addEventListener('click', function() { const doc = new jsPDF(); const region = regionSelect.value; // Title doc.setFontSize(18); doc.text(`Aioxy GHG Emissions Report (${region})`, 105, 15, { align: 'center' }); // Add content... // (Full PDF generation code would go here) doc.save(`Aioxy_GHG_Report_${region}.pdf`); }); // Reset document.getElementById('reset-btn').addEventListener('click', function() { document.querySelectorAll('input[type="number"]').forEach(input => { input.value = ''; }); calculateAll(); }); // Initialize document.addEventListener('DOMContentLoaded', function() { loadCountries(); document.querySelectorAll('input').forEach(input => { input.addEventListener('input', calculateAll); }); document.getElementById('calculate-btn').addEventListener('click', calculateAll); });
